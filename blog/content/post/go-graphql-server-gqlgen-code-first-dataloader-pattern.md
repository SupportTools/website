GraphQL has become the API layer of choice for teams that need to support multiple clients with varying data requirements. Building a production-grade GraphQL server in Go requires careful attention to the N+1 query problem, subscription scaling, and federation architecture. This guide builds a complete implementation using gqlgen that handles all of these production concerns.

```graphql
# graph/schema/schema.graphql

directive @auth(
  requires: Role = USER
) on FIELD_DEFINITION

directive @rateLimit(
  max: Int!
  window: String!
) on FIELD_DEFINITION

enum Role {
  ADMIN
  USER
  READONLY
}

type Query {
  # User queries
  user(id: ID!): User @auth(requires: USER)
  users(
    filter: UserFilter
    pagination: PaginationInput
  ): UserConnection! @auth(requires: USER)
  me: User! @auth(requires: USER)

  # Post queries
  post(id: ID!): Post @auth(requires: USER)
  posts(
    filter: PostFilter
    pagination: PaginationInput
  ): PostConnection! @auth(requires: USER)

  # Search (demonstrates DataLoader for related types)
  search(query: String!, limit: Int): [SearchResult!]!
}

type Mutation {
  # User mutations
  createUser(input: CreateUserInput!): CreateUserPayload! @auth(requires: ADMIN)
  updateUser(id: ID!, input: UpdateUserInput!): UpdateUserPayload! @auth(requires: USER)
  deleteUser(id: ID!): DeleteUserPayload! @auth(requires: ADMIN)

  # Post mutations
  createPost(input: CreatePostInput!): CreatePostPayload! @auth(requires: USER)
  updatePost(id: ID!, input: UpdatePostInput!): UpdatePostPayload! @auth(requires: USER)
  deletePost(id: ID!): DeletePostPayload! @auth(requires: USER)
}

  # Real-time post updates
  postCreated(authorId: ID): Post!
  postUpdated(id: ID!): Post!
  userActivity(userId: ID!): ActivityEvent!

  notifications: Notification! @auth(requires: USER)
}
```

## Section 4: Query Resolvers

```go
// graph/resolvers/query.resolvers.go
package resolvers

import (
    "context"
    "fmt"

    "github.com/yourorg/graphql-service/graph"
    "github.com/yourorg/graphql-service/graph/loaders"
    "github.com/yourorg/graphql-service/graph/model"
    "github.com/yourorg/graphql-service/internal/auth"
)

// queryResolver implements the Query resolvers.
type queryResolver struct{ *graph.Resolver }

func (r *queryResolver) User(ctx context.Context, id string) (*model.User, error) {
    // Use DataLoader to batch database queries
    loader := loaders.For(ctx)
    user, err := loader.UserByID.Load(ctx, id)
    if err != nil {
        return nil, fmt.Errorf("loading user %s: %w", id, err)
    }
    return user, nil
}

func (r *queryResolver) Users(
    ctx context.Context,
    filter *model.UserFilter,
    pagination *model.PaginationInput,
) (*model.UserConnection, error) {
    params := database.UserQueryParams{
        Filter:     convertUserFilter(filter),
        Pagination: convertPagination(pagination),
    }

    result, err := r.DB.QueryUsers(ctx, params)
    if err != nil {
        return nil, fmt.Errorf("querying users: %w", err)
    }

    return buildUserConnection(result), nil
}

func (r *queryResolver) Me(ctx context.Context) (*model.User, error) {
    claims := auth.ClaimsFromContext(ctx)
    if claims == nil {
        return nil, fmt.Errorf("not authenticated")
    }

    loader := loaders.For(ctx)
    return loader.UserByID.Load(ctx, claims.UserID)
}

func (r *queryResolver) Post(ctx context.Context, id string) (*model.Post, error) {
    loader := loaders.For(ctx)
    return loader.PostByID.Load(ctx, id)
}

func (r *queryResolver) Posts(
    ctx context.Context,
    filter *model.PostFilter,
    pagination *model.PaginationInput,
) (*model.PostConnection, error) {
    params := database.PostQueryParams{
        Filter:     convertPostFilter(filter),
        Pagination: convertPagination(pagination),
    }

    result, err := r.DB.QueryPosts(ctx, params)
    if err != nil {
        return nil, fmt.Errorf("querying posts: %w", err)
    }

    return buildPostConnection(result), nil
}

func (r *queryResolver) Search(
    ctx context.Context,
    query string,
    limit *int,
) ([]model.SearchResult, error) {
    maxResults := 20
    if limit != nil && *limit > 0 && *limit < 100 {
        maxResults = *limit
    }

    results, err := r.DB.Search(ctx, query, maxResults)
    if err != nil {
        return nil, fmt.Errorf("search failed: %w", err)
    }

    // Convert to union types
    var searchResults []model.SearchResult
    for _, r := range results {
        switch v := r.(type) {
        case *model.User:
            searchResults = append(searchResults, v)
        case *model.Post:
            searchResults = append(searchResults, v)
        case *model.Tag:
            searchResults = append(searchResults, v)
        }
    }

    return searchResults, nil
}

// Field resolvers for nested types

func (r *postResolver) Author(ctx context.Context, obj *model.Post) (*model.User, error) {
    // This is where N+1 problem occurs WITHOUT DataLoader
    // WITHOUT DataLoader: 1 query per post in a posts list = N+1 queries
    // WITH DataLoader: all authors batched into 1 query
    loader := loaders.For(ctx)
    return loader.UserByID.Load(ctx, obj.AuthorID)
}

func (r *postResolver) Tags(ctx context.Context, obj *model.Post) ([]*model.Tag, error) {
    loader := loaders.For(ctx)
    return loader.TagsByPostID.Load(ctx, obj.ID)
}

func (r *postResolver) Comments(ctx context.Context, obj *model.Post) ([]*model.Comment, error) {
    loader := loaders.For(ctx)
    return loader.CommentsByPostID.Load(ctx, obj.ID)
}

func (r *userResolver) Posts(
    ctx context.Context,
    obj *model.User,
    pagination *model.PaginationInput,
) (*model.PostConnection, error) {
    loader := loaders.For(ctx)
    posts, err := loader.PostsByAuthorID.Load(ctx, obj.ID)
    if err != nil {
        return nil, err
    }
    return buildPostConnection(&database.PostQueryResult{Posts: posts}), nil
}
```

## Section 5: DataLoader Implementation

```go
// graph/loaders/loader.go
package loaders

import (
    "context"
    "fmt"
    "time"

    "github.com/vikstrous/dataloadgen"
    "github.com/yourorg/graphql-service/graph/model"
    "github.com/yourorg/graphql-service/internal/database"
)

// Loaders holds all DataLoader instances for a request.
type Loaders struct {
    UserByID       *dataloadgen.Loader[string, *model.User]
    PostByID       *dataloadgen.Loader[string, *model.Post]
    TagsByPostID   *dataloadgen.Loader[string, []*model.Tag]
    CommentsByPostID *dataloadgen.Loader[string, []*model.Comment]
    PostsByAuthorID  *dataloadgen.Loader[string, []*model.Post]
}

// NewLoaders creates DataLoader instances with batch functions.
func NewLoaders(db database.Repository) *Loaders {
    const (
        wait     = 2 * time.Millisecond
        maxBatch = 100
    )

    return &Loaders{
        UserByID: dataloadgen.NewLoader(
            batchLoadUsers(db),
            dataloadgen.WithWait(wait),
            dataloadgen.WithBatchCapacity(maxBatch),
        ),
        PostByID: dataloadgen.NewLoader(
            batchLoadPosts(db),
            dataloadgen.WithWait(wait),
            dataloadgen.WithBatchCapacity(maxBatch),
        ),
        TagsByPostID: dataloadgen.NewLoader(
            batchLoadTagsByPostID(db),
            dataloadgen.WithWait(wait),
            dataloadgen.WithBatchCapacity(maxBatch),
        ),
        CommentsByPostID: dataloadgen.NewLoader(
            batchLoadCommentsByPostID(db),
            dataloadgen.WithWait(wait),
            dataloadgen.WithBatchCapacity(maxBatch),
        ),
        PostsByAuthorID: dataloadgen.NewLoader(
            batchLoadPostsByAuthorID(db),
            dataloadgen.WithWait(wait),
            dataloadgen.WithBatchCapacity(maxBatch),
        ),
    }
}

// batchLoadUsers fetches multiple users in a single database query.
func batchLoadUsers(db database.Repository) dataloadgen.BatchFunc[string, *model.User] {
    return func(ctx context.Context, ids []string) ([]*model.User, []error) {
        // Single database query for all requested user IDs
        users, err := db.GetUsersByIDs(ctx, ids)
        if err != nil {
            errs := make([]error, len(ids))
            for i := range errs {
                errs[i] = fmt.Errorf("batch load users failed: %w", err)
            }
            return make([]*model.User, len(ids)), errs
        }

        // Build a map for O(1) lookup
        userMap := make(map[string]*model.User, len(users))
        for _, u := range users {
            userMap[u.ID] = u
        }

        // Return in the same order as requested IDs
        result := make([]*model.User, len(ids))
        errs := make([]error, len(ids))
        for i, id := range ids {
            if user, ok := userMap[id]; ok {
                result[i] = user
            } else {
                errs[i] = fmt.Errorf("user %q not found", id)
            }
        }

        return result, errs
    }
}

// batchLoadPosts fetches multiple posts in a single database query.
func batchLoadPosts(db database.Repository) dataloadgen.BatchFunc[string, *model.Post] {
    return func(ctx context.Context, ids []string) ([]*model.Post, []error) {
        posts, err := db.GetPostsByIDs(ctx, ids)
        if err != nil {
            errs := make([]error, len(ids))
            for i := range errs {
                errs[i] = fmt.Errorf("batch load posts failed: %w", err)
            }
            return make([]*model.Post, len(ids)), errs
        }

        postMap := make(map[string]*model.Post, len(posts))
        for _, p := range posts {
            postMap[p.ID] = p
        }

        result := make([]*model.Post, len(ids))
        errs := make([]error, len(ids))
        for i, id := range ids {
            if post, ok := postMap[id]; ok {
                result[i] = post
            } else {
                errs[i] = fmt.Errorf("post %q not found", id)
            }
        }

        return result, errs
    }
}

// batchLoadTagsByPostID fetches tags for multiple posts.
func batchLoadTagsByPostID(db database.Repository) dataloadgen.BatchFunc[string, []*model.Tag] {
    return func(ctx context.Context, postIDs []string) ([][]*model.Tag, []error) {
        // Single query: SELECT * FROM post_tags WHERE post_id IN (...)
        tagsByPost, err := db.GetTagsByPostIDs(ctx, postIDs)
        if err != nil {
            errs := make([]error, len(postIDs))
            for i := range errs {
                errs[i] = fmt.Errorf("batch load tags failed: %w", err)
            }
            return make([][]*model.Tag, len(postIDs)), errs
        }

        result := make([][]*model.Tag, len(postIDs))
        for i, id := range postIDs {
            if tags, ok := tagsByPost[id]; ok {
                result[i] = tags
            } else {
                result[i] = []*model.Tag{} // Empty slice, not error
            }
        }

        return result, make([]error, len(postIDs))
    }
}

// batchLoadCommentsByPostID fetches comments for multiple posts.
func batchLoadCommentsByPostID(db database.Repository) dataloadgen.BatchFunc[string, []*model.Comment] {
    return func(ctx context.Context, postIDs []string) ([][]*model.Comment, []error) {
        commentsByPost, err := db.GetCommentsByPostIDs(ctx, postIDs)
        if err != nil {
            errs := make([]error, len(postIDs))
            for i := range errs {
                errs[i] = fmt.Errorf("batch load comments failed: %w", err)
            }
            return make([][]*model.Comment, len(postIDs)), errs
        }

        result := make([][]*model.Comment, len(postIDs))
        for i, id := range postIDs {
            result[i] = commentsByPost[id]
        }

        return result, make([]error, len(postIDs))
    }
}

// batchLoadPostsByAuthorID fetches posts for multiple authors.
func batchLoadPostsByAuthorID(db database.Repository) dataloadgen.BatchFunc[string, []*model.Post] {
    return func(ctx context.Context, authorIDs []string) ([][]*model.Post, []error) {
        postsByAuthor, err := db.GetPostsByAuthorIDs(ctx, authorIDs)
        if err != nil {
            errs := make([]error, len(authorIDs))
            for i := range errs {
                errs[i] = fmt.Errorf("batch load posts by author failed: %w", err)
            }
            return make([][]*model.Post, len(authorIDs)), errs
        }

        result := make([][]*model.Post, len(authorIDs))
        for i, id := range authorIDs {
            result[i] = postsByAuthor[id]
        }

        return result, make([]error, len(authorIDs))
    }
}
```

### DataLoader Context Middleware

```go
// graph/loaders/middleware.go
package loaders

import (
    "context"
    "net/http"

    "github.com/yourorg/graphql-service/internal/database"
)

type contextKey int

const loadersKey contextKey = iota

// Middleware injects DataLoaders into the request context.
func Middleware(db database.Repository) func(http.Handler) http.Handler {
    return func(next http.Handler) http.Handler {
        return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
            // Create fresh DataLoader instances per request
            // DataLoaders must NOT be shared across requests
            loaders := NewLoaders(db)

            ctx := context.WithValue(r.Context(), loadersKey, loaders)
            next.ServeHTTP(w, r.WithContext(ctx))
        })
    }
}

// For extracts DataLoaders from context.
func For(ctx context.Context) *Loaders {
    loaders, ok := ctx.Value(loadersKey).(*Loaders)
    if !ok {
        panic("DataLoaders not found in context. Did you add the DataLoader middleware?")
    }
    return loaders
}
```

## Section 6: Mutation Resolvers

```go
// graph/resolvers/mutation.resolvers.go
package resolvers

import (
    "context"
    "fmt"
    "strings"

    "github.com/yourorg/graphql-service/graph"
    "github.com/yourorg/graphql-service/graph/model"
    "github.com/yourorg/graphql-service/internal/auth"
    "github.com/yourorg/graphql-service/internal/database"
)

type mutationResolver struct{ *graph.Resolver }

func (r *mutationResolver) CreatePost(
    ctx context.Context,
    input model.CreatePostInput,
) (*model.CreatePostPayload, error) {
    claims := auth.ClaimsFromContext(ctx)
    if claims == nil {
        return nil, fmt.Errorf("not authenticated")
    }

    // Validate input
    var validationErrors []*model.PostError
    if strings.TrimSpace(input.Title) == "" {
        validationErrors = append(validationErrors, &model.PostError{
            Field:   stringPtr("title"),
            Message: "Title cannot be empty",
            Code:    "REQUIRED",
        })
    }
    if len(input.Title) > 500 {
        validationErrors = append(validationErrors, &model.PostError{
            Field:   stringPtr("title"),
            Message: "Title cannot exceed 500 characters",
            Code:    "MAX_LENGTH",
        })
    }
    if len(validationErrors) > 0 {
        return &model.CreatePostPayload{
            Errors: validationErrors,
        }, nil
    }

    // Create post
    post, err := r.DB.CreatePost(ctx, database.CreatePostParams{
        Title:     input.Title,
        Content:   input.Content,
        AuthorID:  claims.UserID,
        Tags:      input.Tags,
        Published: boolValue(input.Published, false),
    })
    if err != nil {
        return nil, fmt.Errorf("creating post: %w", err)
    }

    // Publish to subscriptions
    go func() {
        r.PubSub.Publish("post_created", post)
    }()

    return &model.CreatePostPayload{
        Post:   post,
        Errors: []*model.PostError{},
    }, nil
}

func (r *mutationResolver) UpdatePost(
    ctx context.Context,
    id string,
    input model.UpdatePostInput,
) (*model.UpdatePostPayload, error) {
    claims := auth.ClaimsFromContext(ctx)
    if claims == nil {
        return nil, fmt.Errorf("not authenticated")
    }

    // Load existing post to check ownership
    existingPost, err := r.DB.GetPostByID(ctx, id)
    if err != nil {
        return nil, fmt.Errorf("loading post: %w", err)
    }

    // Authorization check
    if existingPost.AuthorID != claims.UserID && !claims.HasRole("ADMIN") {
        return nil, fmt.Errorf("forbidden: you don't own this post")
    }

    updated, err := r.DB.UpdatePost(ctx, id, database.UpdatePostParams{
        Title:     input.Title,
        Content:   input.Content,
        Tags:      input.Tags,
        Published: input.Published,
    })
    if err != nil {
        return nil, fmt.Errorf("updating post: %w", err)
    }

    // Publish update event
    go func() {
        r.PubSub.Publish(fmt.Sprintf("post_updated_%s", id), updated)
    }()

    return &model.UpdatePostPayload{
        Post:   updated,
        Errors: []*model.PostError{},
    }, nil
}

func stringPtr(s string) *string { return &s }
func boolValue(b *bool, def bool) bool {
    if b == nil {
        return def
    }
    return *b
}
```

```go
func (r *subscriptionResolver) PostCreated(
    ctx context.Context,
    authorID *string,
) (<-chan *model.Post, error) {
    ch := make(chan *model.Post, 1)

    topic := "post_created"
    if authorID != nil {
        topic = fmt.Sprintf("post_created_author_%s", *authorID)
    }

    sub, err := r.PubSub.Subscribe(ctx, topic)
    if err != nil {
        return nil, fmt.Errorf("subscribing to post events: %w", err)
    }

    go func() {
        defer close(ch)
        defer sub.Unsubscribe()

        for {
            select {
            case <-ctx.Done():
                return
            case event, ok := <-sub.Channel():
                if !ok {
                    return
                }

                post, ok := event.(*model.Post)
                if !ok {
                    continue
                }

                // Filter by author if specified
                if authorID != nil && post.AuthorID != *authorID {
                    continue
                }

                select {
                case ch <- post:
                case <-ctx.Done():
                    return
                case <-time.After(5 * time.Second):
                    // Slow consumer, skip this event
                    r.Logger.Warn("subscription consumer too slow, dropping event")
                }
            }
        }
    }()

    return ch, nil
}
```

```go
// internal/auth/middleware.go
package auth

import (
    "context"
    "net/http"
    "strings"

    "github.com/golang-jwt/jwt/v5"
    "go.uber.org/zap"
)

type contextKey int

const claimsContextKey contextKey = iota

// Claims represents the JWT claims.
type Claims struct {
    jwt.RegisteredClaims
    UserID string   `json:"uid"`
    Email  string   `json:"email"`
    Roles  []string `json:"roles"`
}

// HasRole checks if the claims include the given role.
func (c *Claims) HasRole(role string) bool {
    for _, r := range c.Roles {
        if r == role {
            return true
        }
    }
    return false
}

// Middleware extracts and validates JWT from requests.
func Middleware(jwtSecret []byte, logger *zap.Logger) func(http.Handler) http.Handler {
    return func(next http.Handler) http.Handler {
        return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
            token := extractToken(r)
            if token == "" {
                // Allow unauthenticated requests (auth is enforced per-field)
                next.ServeHTTP(w, r)
                return
            }

            claims, err := validateToken(token, jwtSecret)
            if err != nil {
                logger.Debug("invalid JWT", zap.Error(err))
                // Don't reject; let field-level auth handle it
                next.ServeHTTP(w, r)
                return
            }

            ctx := context.WithValue(r.Context(), claimsContextKey, claims)
            next.ServeHTTP(w, r.WithContext(ctx))
        })
    }
}

func extractToken(r *http.Request) string {
    // Check Authorization header
    authHeader := r.Header.Get("Authorization")
    if authHeader != "" {
        return strings.TrimPrefix(authHeader, "Bearer ")
    }

    // Check cookie (for WebSocket connections)
    if cookie, err := r.Cookie("auth_token"); err == nil {
        return cookie.Value
    }

    // Check query parameter (last resort, for WebSocket upgrade requests)
    return r.URL.Query().Get("token")
}

func validateToken(tokenString string, secret []byte) (*Claims, error) {
    claims := &Claims{}
    _, err := jwt.ParseWithClaims(tokenString, claims,
        func(t *jwt.Token) (interface{}, error) {
            if _, ok := t.Method.(*jwt.SigningMethodHMAC); !ok {
                return nil, jwt.ErrSignatureInvalid
            }
            return secret, nil
        },
        jwt.WithExpirationRequired(),
    )
    if err != nil {
        return nil, err
    }
    return claims, nil
}

// ClaimsFromContext extracts JWT claims from context.
func ClaimsFromContext(ctx context.Context) *Claims {
    claims, _ := ctx.Value(claimsContextKey).(*Claims)
    return claims
}
```

### @auth Directive Implementation

```go
// graph/directives/auth.go
package directives

import (
    "context"
    "fmt"

    "github.com/99designs/gqlgen/graphql"
    "github.com/yourorg/graphql-service/graph/model"
    "github.com/yourorg/graphql-service/internal/auth"
)

// AuthDirective implements the @auth directive.
func AuthDirective(ctx context.Context, obj interface{}, next graphql.Resolver, requires model.Role) (interface{}, error) {
    claims := auth.ClaimsFromContext(ctx)
    if claims == nil {
        return nil, fmt.Errorf("UNAUTHENTICATED: login required")
    }

    switch requires {
    case model.RoleAdmin:
        if !claims.HasRole("ADMIN") {
            return nil, fmt.Errorf("FORBIDDEN: admin role required")
        }
    case model.RoleUser:
        if !claims.HasRole("USER") && !claims.HasRole("ADMIN") {
            return nil, fmt.Errorf("FORBIDDEN: user role required")
        }
    case model.RoleReadonly:
        // Any authenticated user passes
    }

    return next(ctx)
}
```

```go
// cmd/server/main.go
package main

import (
    "context"
    "net/http"
    "os"
    "os/signal"
    "syscall"
    "time"

    "github.com/99designs/gqlgen/graphql/handler"
    "github.com/99designs/gqlgen/graphql/handler/extension"
    "github.com/99designs/gqlgen/graphql/handler/lru"
    "github.com/99designs/gqlgen/graphql/handler/transport"
    "github.com/99designs/gqlgen/graphql/playground"
    "github.com/gorilla/websocket"
    "github.com/prometheus/client_golang/prometheus/promhttp"
    "go.uber.org/zap"

    "github.com/yourorg/graphql-service/graph"
    "github.com/yourorg/graphql-service/graph/directives"
    "github.com/yourorg/graphql-service/graph/generated"
    "github.com/yourorg/graphql-service/graph/loaders"
    "github.com/yourorg/graphql-service/internal/auth"
    "github.com/yourorg/graphql-service/internal/database"
    "github.com/yourorg/graphql-service/internal/pubsub"
)

func main() {
    logger, _ := zap.NewProduction()
    defer logger.Sync() //nolint:errcheck

    // Initialize dependencies
    db, err := database.New(os.Getenv("DATABASE_URL"))
    if err != nil {
        logger.Fatal("failed to connect to database", zap.Error(err))
    }

    ps := pubsub.NewRedis(os.Getenv("REDIS_URL"))

    // Build resolver
    resolver := &graph.Resolver{
        DB:     db,
        PubSub: ps,
        Logger: logger,
    }

    // Build GraphQL schema config
    config := generated.Config{
        Resolvers: resolver,
        Directives: generated.DirectiveRoot{
            Auth: directives.AuthDirective,
        },
    }

    schema := generated.NewExecutableSchema(config)

    // Create handler with all transports
    srv := handler.New(schema)

    // WebSocket transport for subscriptions
    srv.AddTransport(transport.Websocket{
        KeepAlivePingInterval: 10 * time.Second,
        Upgrader: websocket.Upgrader{
            CheckOrigin: func(r *http.Request) bool {
                // In production, validate origin
                return true
            },
            HandshakeTimeout: 15 * time.Second,
        },
        InitFunc: func(ctx context.Context, initPayload transport.InitPayload) (context.Context, *transport.InitPayload, error) {
            // Extract auth token from WebSocket init payload
            token, _ := initPayload["authToken"].(string)
            if token != "" {
                jwtSecret := []byte(os.Getenv("JWT_SECRET"))
                if claims, err := auth.ValidateToken(token, jwtSecret); err == nil {
                    ctx = auth.WithClaims(ctx, claims)
                }
            }
            return ctx, &initPayload, nil
        },
    })
    srv.AddTransport(transport.Options{})
    srv.AddTransport(transport.GET{})
    srv.AddTransport(transport.POST{})
    srv.AddTransport(transport.MultipartForm{})

    // Query complexity limiting
    srv.Use(extension.FixedComplexityLimit(1000))

    // Automatic persisted queries
    srv.Use(extension.AutomaticPersistedQuery{
        Cache: lru.New[string](100),
    })

    // Introspection (disable in production)
    if os.Getenv("ENV") != "production" {
        srv.Use(extension.Introspection{})
    }

    // HTTP mux
    mux := http.NewServeMux()

    // GraphQL endpoint with middleware chain
    graphqlHandler := auth.Middleware(
        []byte(os.Getenv("JWT_SECRET")),
        logger,
    )(loaders.Middleware(db)(srv))

    mux.Handle("/graphql", graphqlHandler)
    mux.Handle("/metrics", promhttp.Handler())

    // Playground (development only)
    if os.Getenv("ENV") != "production" {
        mux.Handle("/", playground.Handler("GraphQL Playground", "/graphql"))
    }

    httpSrv := &http.Server{
        Addr:         ":8080",
        Handler:      mux,
        ReadTimeout:  30 * time.Second,
        WriteTimeout: 30 * time.Second,
        IdleTimeout:  120 * time.Second,
    }

    // Graceful shutdown
    sigCh := make(chan os.Signal, 1)
    signal.Notify(sigCh, syscall.SIGTERM, syscall.SIGINT)

    go func() {
        <-sigCh
        logger.Info("shutting down server")
        ctx, cancel := context.WithTimeout(context.Background(), 30*time.Second)
        defer cancel()
        if err := httpSrv.Shutdown(ctx); err != nil {
            logger.Error("server shutdown error", zap.Error(err))
        }
    }()

    logger.Info("GraphQL server starting", zap.String("addr", httpSrv.Addr))
    if err := httpSrv.ListenAndServe(); err != http.ErrServerClosed {
        logger.Fatal("server error", zap.Error(err))
    }
}
```

The DataLoader pattern is what separates a performant GraphQL API from one that collapses under load. Without it, a query for 100 posts with their authors generates 101 database queries. With properly implemented DataLoaders, that same query requires exactly 2 queries—one for posts, one for the batched author lookups. Combined with query complexity limits and persisted queries, this architecture handles production traffic at scale.
