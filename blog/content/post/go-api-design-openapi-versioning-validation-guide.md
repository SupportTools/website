A REST API is a contract. Changing it breaks consumers. This guide covers the full API lifecycle in Go: versioning strategies (URL path, header, content negotiation), auto-generating OpenAPI 3.0 specs from code, request validation middleware, response envelope patterns, error schema design, and the backward compatibility rules that let you evolve APIs without breaking clients.

# Go API Design: RESTful Versioning, OpenAPI Generation, Request Validation, and Backward Compatibility

## Section 1: API Versioning Strategies

Three main approaches, each with tradeoffs:

| Strategy | Example | Pro | Con |
|----------|---------|-----|-----|
| URL path | `/api/v1/users` | Explicit, cacheable | URL pollution |
| Header | `Accept-Version: v1` | Clean URLs | Harder to test |
| Content type | `Accept: application/vnd.myapi.v1+json` | RFC-compliant | Complex |

URL path versioning is the most practical for public APIs. Header versioning works well for internal APIs:

```go
// router/router.go
package router

import (
    "net/http"

    "github.com/go-chi/chi/v5"
    "github.com/go-chi/chi/v5/middleware"

    v1 "github.com/myorg/api/handlers/v1"
    v2 "github.com/myorg/api/handlers/v2"
)

func New() http.Handler {
    r := chi.NewRouter()

    // Global middleware
    r.Use(middleware.RequestID)
    r.Use(middleware.RealIP)
    r.Use(middleware.Logger)
    r.Use(middleware.Recoverer)
    r.Use(middleware.Compress(5))

    // API v1
    r.Route("/api/v1", func(r chi.Router) {
        r.Use(VersionMiddleware("v1"))
        r.Mount("/users", v1.UserRoutes())
        r.Mount("/orders", v1.OrderRoutes())
    })

    // API v2 — additive changes only
    r.Route("/api/v2", func(r chi.Router) {
        r.Use(VersionMiddleware("v2"))
        r.Mount("/users", v2.UserRoutes())
        r.Mount("/orders", v2.OrderRoutes())
    })

    // Header-based versioning for internal services
    r.Route("/internal/api", func(r chi.Router) {
        r.Use(HeaderVersionMiddleware)
        r.Mount("/events", internalEventRoutes())
    })

    // OpenAPI spec served from /docs
    r.Get("/docs/openapi.yaml", serveOpenAPISpec)
    r.Get("/docs/openapi.json", serveOpenAPISpecJSON)

    return r
}

// VersionMiddleware injects the API version into context.
func VersionMiddleware(version string) func(http.Handler) http.Handler {
    return func(next http.Handler) http.Handler {
        return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
            ctx := context.WithValue(r.Context(), contextKeyVersion, version)
            w.Header().Set("X-API-Version", version)
            next.ServeHTTP(w, r.WithContext(ctx))
        })
    }
}

// HeaderVersionMiddleware reads version from Accept-Version header.
func HeaderVersionMiddleware(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        version := r.Header.Get("Accept-Version")
        if version == "" {
            version = "v1"  // default to v1
        }
        // Validate version
        validVersions := map[string]bool{"v1": true, "v2": true, "v3": true}
        if !validVersions[version] {
            writeError(w, http.StatusBadRequest,
                "INVALID_VERSION",
                "Unsupported API version: "+version,
                nil)
            return
        }
        ctx := context.WithValue(r.Context(), contextKeyVersion, version)
        next.ServeHTTP(w, r.WithContext(ctx))
    })
}
```

```go
// handlers/v2/user.go
package v2

import (
    "net/http"

    "github.com/go-chi/chi/v5"
    "github.com/myorg/api/internal/service"
    "github.com/myorg/api/models"
)

// CreateUser creates a new user account.
//
// @Summary      Create user
// @Description  Create a new user account with the provided details
// @Tags         users
// @Accept       json
// @Produce      json
// @Param        request  body      models.CreateUserRequest  true  "User creation request"
// @Success      201      {object}  models.UserResponse
// @Failure      400      {object}  models.ErrorResponse
// @Failure      409      {object}  models.ErrorResponse  "User already exists"
// @Failure      422      {object}  models.ValidationErrorResponse
// @Failure      500      {object}  models.ErrorResponse
// @Security     BearerAuth
// @Router       /users [post]
func (h *UserHandler) CreateUser(w http.ResponseWriter, r *http.Request) {
    var req models.CreateUserRequest
    if err := decodeAndValidate(r, &req); err != nil {
        writeValidationError(w, err)
        return
    }

    user, err := h.svc.CreateUser(r.Context(), &req)
    if err != nil {
        writeServiceError(w, err)
        return
    }

    writeJSON(w, http.StatusCreated, models.UserResponse{
        Data: user,
        Meta: models.Meta{
            Version:   "v2",
            RequestID: requestIDFromContext(r.Context()),
        },
    })
}

// GetUser retrieves a user by ID.
//
// @Summary      Get user
// @Description  Retrieve user details by user ID
// @Tags         users
// @Produce      json
// @Param        id   path      string  true  "User ID (UUID)"
// @Param        expand  query  string  false  "Comma-separated list of fields to expand"  Enums(profile,orders,preferences)
// @Success      200  {object}  models.UserResponse
// @Failure      404  {object}  models.ErrorResponse
// @Security     BearerAuth
// @Router       /users/{id} [get]
func (h *UserHandler) GetUser(w http.ResponseWriter, r *http.Request) {
    userID := chi.URLParam(r, "id")
    expand := parseExpandParam(r.URL.Query().Get("expand"))

    user, err := h.svc.GetUser(r.Context(), userID, expand)
    if err != nil {
        writeServiceError(w, err)
        return
    }

    writeJSON(w, http.StatusOK, models.UserResponse{
        Data: user,
        Meta: models.Meta{
            Version:   "v2",
            RequestID: requestIDFromContext(r.Context()),
        },
    })
}
```

## Section 3: Request Validation Middleware

```go
// middleware/validation.go
package middleware

import (
    "encoding/json"
    "fmt"
    "io"
    "net/http"
    "reflect"
    "strings"
    "unicode/utf8"

    "github.com/go-playground/validator/v10"
)

var validate *validator.Validate

func init() {
    validate = validator.New()

    // Register custom tag for reading JSON field names in error messages
    validate.RegisterTagNameFunc(func(fld reflect.StructField) string {
        name := strings.SplitN(fld.Tag.Get("json"), ",", 2)[0]
        if name == "-" {
            return ""
        }
        return name
    })

    // Custom validators
    validate.RegisterValidation("slug", validateSlug)
    validate.RegisterValidation("semver", validateSemver)
    validate.RegisterValidation("no_html", validateNoHTML)
}

// DecodeAndValidate decodes the JSON body and validates the struct.
func DecodeAndValidate(r *http.Request, dst interface{}) error {
    // Limit body size to prevent DoS
    body := io.LimitReader(r.Body, 1<<20) // 1MB

    dec := json.NewDecoder(body)
    dec.DisallowUnknownFields()

    if err := dec.Decode(dst); err != nil {
        switch {
        case err == io.EOF:
            return &ValidationError{Message: "request body is empty"}
        case strings.Contains(err.Error(), "unknown field"):
            field := extractUnknownField(err.Error())
            return &ValidationError{
                Message: "unknown field in request",
                Fields: []FieldError{{
                    Field:   field,
                    Message: "field not allowed",
                    Code:    "unknown_field",
                }},
            }
        default:
            return &ValidationError{Message: "invalid JSON: " + err.Error()}
        }
    }

    if err := validate.Struct(dst); err != nil {
        return buildValidationError(err)
    }

    return nil
}

// ValidationError represents a structured validation failure.
type ValidationError struct {
    Message string
    Fields  []FieldError
}

type FieldError struct {
    Field   string `json:"field"`
    Message string `json:"message"`
    Code    string `json:"code"`
    Value   string `json:"value,omitempty"`
}

func (e *ValidationError) Error() string { return e.Message }

func buildValidationError(err error) *ValidationError {
    ve := &ValidationError{Message: "request validation failed"}

    for _, fe := range err.(validator.ValidationErrors) {
        ve.Fields = append(ve.Fields, FieldError{
            Field:   fe.Field(),
            Message: fieldErrorMessage(fe),
            Code:    fe.Tag(),
            Value:   fmt.Sprintf("%v", fe.Value()),
        })
    }
    return ve
}

func fieldErrorMessage(fe validator.FieldError) string {
    switch fe.Tag() {
    case "required":
        return "field is required"
    case "min":
        return fmt.Sprintf("must be at least %s characters", fe.Param())
    case "max":
        return fmt.Sprintf("must not exceed %s characters", fe.Param())
    case "email":
        return "must be a valid email address"
    case "url":
        return "must be a valid URL"
    case "uuid":
        return "must be a valid UUID"
    case "oneof":
        return "must be one of: " + strings.ReplaceAll(fe.Param(), " ", ", ")
    case "slug":
        return "must contain only lowercase letters, numbers, and hyphens"
    case "no_html":
        return "must not contain HTML tags"
    default:
        return fmt.Sprintf("failed validation: %s", fe.Tag())
    }
}

func validateSlug(fl validator.FieldLevel) bool {
    s := fl.Field().String()
    for _, c := range s {
        if !((c >= 'a' && c <= 'z') || (c >= '0' && c <= '9') || c == '-') {
            return false
        }
    }
    return len(s) > 0
}

func validateNoHTML(fl validator.FieldLevel) bool {
    s := fl.Field().String()
    return !strings.ContainsAny(s, "<>") && !strings.Contains(s, "&lt;")
}

func validateSemver(fl validator.FieldLevel) bool {
    s := fl.Field().String()
    parts := strings.Split(strings.TrimPrefix(s, "v"), ".")
    return len(parts) == 3
}

func extractUnknownField(errMsg string) string {
    // json: unknown field "fieldName"
    start := strings.Index(errMsg, `"`)
    end := strings.LastIndex(errMsg, `"`)
    if start >= 0 && end > start {
        return errMsg[start+1 : end]
    }
    return "unknown"
}
```

## Section 4: Request/Response Models with Backward Compatibility

```go
// models/user.go
package models

import (
    "time"
)

// CreateUserRequest v2 — extends v1 by adding phone field
type CreateUserRequest struct {
    Email     string  `json:"email"     validate:"required,email,max=254"`
    FirstName string  `json:"first_name" validate:"required,min=1,max=100,no_html"`
    LastName  string  `json:"last_name"  validate:"required,min=1,max=100,no_html"`
    Phone     *string `json:"phone"     validate:"omitempty,min=7,max=20"`
    // v2 additions — optional for backward compatibility
    Locale    *string `json:"locale"    validate:"omitempty,oneof=en-US en-GB fr-FR de-DE"`
    Timezone  *string `json:"timezone"  validate:"omitempty,max=50"`
}

// User is the internal domain model — never exposed directly.

    Locale    string
```

## Section 5: Response Envelope and Error Schema

```go
// models/response.go
package models

// Envelope wraps all API responses for consistent structure.
type Envelope[T any] struct {
    Data    T           `json:"data"`
    Meta    Meta        `json:"meta"`
    Links   *Links      `json:"links,omitempty"`
}

type Meta struct {
    RequestID string `json:"request_id"`
    Version   string `json:"api_version"`
    Timestamp int64  `json:"timestamp"`
}

// PaginatedEnvelope adds pagination to the response.
type PaginatedEnvelope[T any] struct {
    Data       []T        `json:"data"`
    Meta       Meta       `json:"meta"`
    Pagination Pagination `json:"pagination"`
    Links      Links      `json:"links"`
}

type Pagination struct {
    Page       int `json:"page"`
    PageSize   int `json:"page_size"`
    TotalItems int `json:"total_items"`
    TotalPages int `json:"total_pages"`
    HasMore    bool `json:"has_more"`
}

type Links struct {
    Self  string `json:"self,omitempty"`
    Next  string `json:"next,omitempty"`
    Prev  string `json:"prev,omitempty"`
    First string `json:"first,omitempty"`
    Last  string `json:"last,omitempty"`
}

// ErrorResponse is the standard error envelope.
type ErrorResponse struct {
    Error ErrorBody `json:"error"`
    Meta  Meta      `json:"meta"`
}

type ErrorBody struct {
    Code    string        `json:"code"`
    Message string        `json:"message"`
    Details []ErrorDetail `json:"details,omitempty"`
    TraceID string        `json:"trace_id,omitempty"`
}

type ErrorDetail struct {
    Field   string `json:"field,omitempty"`
    Code    string `json:"code"`
    Message string `json:"message"`
}

// Standard error codes
const (
    ErrCodeValidation     = "VALIDATION_ERROR"
    ErrCodeNotFound       = "NOT_FOUND"
    ErrCodeUnauthorized   = "UNAUTHORIZED"
    ErrCodeForbidden      = "FORBIDDEN"
    ErrCodeConflict       = "CONFLICT"
    ErrCodeRateLimited    = "RATE_LIMITED"
    ErrCodeInternalError  = "INTERNAL_ERROR"
    ErrCodeServiceUnavail = "SERVICE_UNAVAILABLE"
)
```

```go
// handlers/helpers.go
package handlers

import (
    "encoding/json"
    "net/http"
    "time"

    "github.com/myorg/api/models"
)

func WriteJSON[T any](w http.ResponseWriter, status int, data T, r *http.Request) {
    resp := models.Envelope[T]{
        Data: data,
        Meta: models.Meta{
            RequestID: requestIDFromContext(r.Context()),
            Version:   versionFromContext(r.Context()),
            Timestamp: time.Now().UnixMilli(),
        },
    }
    w.Header().Set("Content-Type", "application/json")
    w.WriteHeader(status)
    json.NewEncoder(w).Encode(resp)
}

func WritePaginated[T any](w http.ResponseWriter, status int, items []T,
    page, pageSize, total int, r *http.Request) {
    totalPages := (total + pageSize - 1) / pageSize
    baseURL := r.URL.Path

    resp := models.PaginatedEnvelope[T]{
        Data: items,
        Meta: models.Meta{
            RequestID: requestIDFromContext(r.Context()),
            Version:   versionFromContext(r.Context()),
            Timestamp: time.Now().UnixMilli(),
        },
        Pagination: models.Pagination{
            Page:       page,
            PageSize:   pageSize,
            TotalItems: total,
            TotalPages: totalPages,
            HasMore:    page < totalPages,
        },
        Links: models.Links{
            Self: fmt.Sprintf("%s?page=%d&page_size=%d", baseURL, page, pageSize),
        },
    }
    if page > 1 {
        resp.Links.Prev = fmt.Sprintf("%s?page=%d&page_size=%d", baseURL, page-1, pageSize)
    }
    if page < totalPages {
        resp.Links.Next = fmt.Sprintf("%s?page=%d&page_size=%d", baseURL, page+1, pageSize)
    }

    w.Header().Set("Content-Type", "application/json")
    w.WriteHeader(status)
    json.NewEncoder(w).Encode(resp)
}

func WriteError(w http.ResponseWriter, status int, code, message string, details []models.ErrorDetail, r *http.Request) {
    resp := models.ErrorResponse{
        Error: models.ErrorBody{
            Code:    code,
            Message: message,
            Details: details,
        },
        Meta: models.Meta{
            RequestID: requestIDFromContext(r.Context()),
            Version:   versionFromContext(r.Context()),
            Timestamp: time.Now().UnixMilli(),
        },
    }
    w.Header().Set("Content-Type", "application/json")
    w.WriteHeader(status)
    json.NewEncoder(w).Encode(resp)
}
```

## Section 6: Rate Limiting and Throttling

```go
// middleware/ratelimit.go
package middleware

import (
    "net/http"
    "time"

    "github.com/redis/go-redis/v9"
    "golang.org/x/time/rate"
)

// PerClientRateLimiter uses Redis sliding window for distributed rate limiting.
type PerClientRateLimiter struct {
    rdb      *redis.Client
    limit    int     // requests per window
    window   time.Duration
}

func (rl *PerClientRateLimiter) Middleware(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        clientID := extractClientID(r)
        key := "ratelimit:" + clientID

        now := time.Now().UnixMilli()
        windowStart := now - rl.window.Milliseconds()

        pipe := rl.rdb.Pipeline()
        // Sliding window using sorted set
        pipe.ZRemRangeByScore(r.Context(), key, "0", fmt.Sprintf("%d", windowStart))
        pipe.ZCard(r.Context(), key)
        pipe.ZAdd(r.Context(), key, redis.Z{Score: float64(now), Member: now})
        pipe.Expire(r.Context(), key, rl.window*2)

        results, err := pipe.Exec(r.Context())
        if err != nil {
            // Fail open — allow request if Redis is unavailable
            next.ServeHTTP(w, r)
            return
        }

        count := results[1].(*redis.IntCmd).Val()
        remaining := int64(rl.limit) - count

        w.Header().Set("X-RateLimit-Limit", fmt.Sprintf("%d", rl.limit))
        w.Header().Set("X-RateLimit-Remaining", fmt.Sprintf("%d", max(0, remaining)))
        w.Header().Set("X-RateLimit-Reset", fmt.Sprintf("%d", (now+rl.window.Milliseconds())/1000))

        if count > int64(rl.limit) {
            WriteError(w, http.StatusTooManyRequests,
                models.ErrCodeRateLimited,
                fmt.Sprintf("Rate limit exceeded. Try again in %d seconds.", rl.window/time.Second),
                nil, r)
            return
        }

        next.ServeHTTP(w, r)
    })
}

func extractClientID(r *http.Request) string {
    // Use API key if authenticated
    if apiKey := r.Header.Get("X-API-Key"); apiKey != "" {
        return "key:" + hashKey(apiKey)
    }
    // Fall back to IP
    return "ip:" + realIP(r)
}
```

## Section 8: Deprecation and Sunset Headers

```go
// middleware/deprecation.go
package middleware

import (
    "net/http"
    "time"
)

// DeprecationNotice adds Deprecation and Sunset headers to deprecated endpoint responses.
// See RFC 8594 and draft-ietf-httpapi-deprecation-header.
func DeprecationNotice(deprecatedDate, sunsetDate time.Time, link string) func(http.Handler) http.Handler {
    deprecatedHeader := deprecatedDate.UTC().Format(http.TimeFormat)
    sunsetHeader := sunsetDate.UTC().Format(http.TimeFormat)

    return func(next http.Handler) http.Handler {
        return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
            w.Header().Set("Deprecation", deprecatedHeader)
            w.Header().Set("Sunset", sunsetHeader)
            if link != "" {
                w.Header().Set("Link", `<`+link+`>; rel="deprecation"`)
            }
            next.ServeHTTP(w, r)
        })
    }
}
```

## Section 9: Backward Compatibility Rules

These rules prevent breaking existing clients:

```markdown
SAFE CHANGES (non-breaking):
✓ Adding new optional fields to responses
✓ Adding new optional fields to requests
✓ Adding new API endpoints
✓ Adding new enum values to fields where client ignores unknown values
✓ Making a required request field optional
✓ Adding new HTTP methods to existing endpoints
✓ Adding new query parameters with defaults

BREAKING CHANGES (require new version):
✗ Removing fields from responses
✗ Renaming fields
✗ Changing field types (e.g., string -> integer)
✗ Making optional request fields required
✗ Removing enum values that clients may send
✗ Changing HTTP status codes for existing scenarios
✗ Changing URL path structure
✗ Changing authentication/authorization requirements
✗ Changing error code strings
```

```go
// tools/breaking-change-detector.go
// Run in CI to catch breaking changes between API spec versions
package main

import (
    "fmt"
    "os"

    "github.com/oasdiff/oasdiff/diff"
    "github.com/oasdiff/oasdiff/load"
)

func main() {
    baseSpec := os.Args[1]    // previous OpenAPI spec
    revisionSpec := os.Args[2] // new OpenAPI spec

    loader := load.NewLoader()
    base, _ := loader.LoadFromPath(baseSpec)
    revision, _ := loader.LoadFromPath(revisionSpec)

    diffReport, err := diff.Get(diff.NewConfig(), base, revision)
    if err != nil {
        fmt.Fprintf(os.Stderr, "diff error: %v\n", err)
        os.Exit(1)
    }

    if diffReport.Empty() {
        fmt.Println("No changes detected")
        os.Exit(0)
    }

    // Check for breaking changes
    breaking := diffReport.GetBreakingChanges()
    if len(breaking) > 0 {
        fmt.Fprintln(os.Stderr, "BREAKING CHANGES DETECTED:")
        for _, change := range breaking {
            fmt.Fprintf(os.Stderr, "  - %s\n", change)
        }
        os.Exit(1)
    }

    fmt.Println("No breaking changes. Diff summary:")
    fmt.Println(diffReport.String())
}
```
