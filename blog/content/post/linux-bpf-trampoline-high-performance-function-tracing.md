BPF trampolines represent the most significant advancement in Linux function tracing since kprobes. Where kprobes trap into the kernel via int3 instructions—causing context switches and significant overhead—BPF trampolines inline themselves directly into the function prologue with near-zero overhead. Understanding this mechanism is essential for anyone building high-frequency observability tooling or performance-critical tracing infrastructure.

# Linux BPF Trampoline: High-Performance Function Tracing without Overhead

## Section 1: The Problem with kprobes

Before examining BPF trampolines, understanding why they were created requires examining the overhead profile of their predecessor.

### kprobe Execution Path

```
Function call:
  1. CPU executes JMP to kprobe handler
  2. Save registers (expensive!)
  3. Execute pre_handler (eBPF program via kprobe)
  4. Single-step original instruction
  5. Execute post_handler
  6. Restore registers
  7. Continue original execution

Cost: ~100-300ns per invocation on modern hardware
```

For functions called millions of times per second, this overhead becomes significant enough to alter the behavior being measured—the observer effect in action.

## Section 2: BPF Trampoline Architecture

### How Trampolines Work

A BPF trampoline is a small piece of machine code (an "island") generated by the BPF JIT compiler and inserted directly between the function call site and the function prologue.

```
Without BPF trampoline:
  caller -> [function prologue] -> function body

With BPF trampoline (fentry):
  caller -> [BPF trampoline] -> [function prologue] -> function body

BPF trampoline assembly (x86_64, simplified):
  PUSH %rbp                    ; save frame pointer
  MOV %rsp, %rbp              ; set up stack frame
  PUSH original_args...        ; save function arguments
  CALL bpf_prog                ; call BPF program (inlined, no context switch)
  POP original_args...         ; restore arguments
  POP %rbp                     ; restore frame pointer
  JMP [original_function]      ; jump to original function
```

The critical difference: no int3 trap, no context switch, no single-step execution. The BPF program runs in the same execution context as the traced function.

### Overhead Comparison

| Mechanism | Overhead | Per-call Cost |
|-----------|----------|---------------|
| kprobe | High | 100-300ns |
| uprobe | Medium-High | 80-200ns |
| BPF fentry/fexit | Minimal | 5-15ns |
| perf_event (PMU) | Minimal | 2-5ns |

## Section 3: fentry and fexit Programs

### fentry: Function Entry Tracing

```c
// fentry_example.bpf.c
#include "vmlinux.h"
#include <bpf/bpf_helpers.h>
#include <bpf/bpf_tracing.h>
#include <bpf/bpf_core_read.h>

// Per-CPU map for storing timing data
struct {
    __uint(type, BPF_MAP_TYPE_PERCPU_HASH);
    __uint(max_entries, 10240);
    __type(key, u64);        // PID
    __type(value, u64);      // entry timestamp
} start_times SEC(".maps");

// Statistics map
struct {
    __uint(type, BPF_MAP_TYPE_ARRAY);
    __uint(max_entries, 1);
    __type(key, u32);
    __type(value, struct stats);
} stats_map SEC(".maps");

struct stats {
    __u64 count;
    __u64 total_ns;
    __u64 max_ns;
    __u64 min_ns;
};

// fentry probe - called on every entry to tcp_sendmsg
// Arguments match the original kernel function signature
SEC("fentry/tcp_sendmsg")
int BPF_PROG(fentry_tcp_sendmsg, struct sock *sk, struct msghdr *msg, size_t size)
{
    u64 pid_tgid = bpf_get_current_pid_tgid();
    u64 ts = bpf_ktime_get_ns();

    // Store entry time keyed by PID
    bpf_map_update_elem(&start_times, &pid_tgid, &ts, BPF_ANY);

    return 0;
}

// fexit probe - called after tcp_sendmsg returns
SEC("fexit/tcp_sendmsg")
int BPF_PROG(fexit_tcp_sendmsg, struct sock *sk, struct msghdr *msg, size_t size, int ret)
{
    u64 pid_tgid = bpf_get_current_pid_tgid();
    u64 *start_ts = bpf_map_lookup_elem(&start_times, &pid_tgid);
    if (!start_ts)
        return 0;

    u64 duration = bpf_ktime_get_ns() - *start_ts;
    bpf_map_delete_elem(&start_times, &pid_tgid);

    // Update statistics
    u32 key = 0;
    struct stats *s = bpf_map_lookup_elem(&stats_map, &key);
    if (!s) {
        struct stats new_stats = {
            .count = 1,
            .total_ns = duration,
            .max_ns = duration,
            .min_ns = duration,
        };
        bpf_map_update_elem(&stats_map, &key, &new_stats, BPF_NOEXIST);
        return 0;
    }

    // Atomic update (per-CPU map avoids contention)
    __sync_fetch_and_add(&s->count, 1);
    __sync_fetch_and_add(&s->total_ns, duration);

    // Non-atomic max/min (acceptable approximation for metrics)
    if (duration > s->max_ns)
        s->max_ns = duration;
    if (duration < s->min_ns || s->min_ns == 0)
        s->min_ns = duration;

    return 0;
}
```

### Userspace Loader with libbpf

```c
// fentry_example.c - userspace loader
#include <stdio.h>
#include <stdlib.h>
#include <signal.h>
#include <unistd.h>
#include <time.h>
#include <bpf/libbpf.h>
#include "fentry_example.skel.h"

static volatile bool running = true;

void sig_handler(int sig) {
    running = false;
}

int main(int argc, char **argv) {
    struct fentry_example_bpf *skel;
    int err;

    // Set up signal handlers
    signal(SIGINT, sig_handler);
    signal(SIGTERM, sig_handler);

    // Load and verify BPF application
    skel = fentry_example_bpf__open();
    if (!skel) {
        fprintf(stderr, "Failed to open BPF skeleton\n");
        return 1;
    }

    // Load into kernel
    err = fentry_example_bpf__load(skel);
    if (err) {
        fprintf(stderr, "Failed to load BPF skeleton: %d\n", err);
        goto cleanup;
    }

    // Attach fentry/fexit programs
    err = fentry_example_bpf__attach(skel);
    if (err) {
        fprintf(stderr, "Failed to attach BPF programs: %d\n", err);
        goto cleanup;
    }

    printf("Tracing tcp_sendmsg with BPF trampoline... Hit Ctrl+C to stop\n");

    while (running) {
        sleep(1);

        // Read statistics
        __u32 key = 0;
        struct stats s = {};
        if (bpf_map__lookup_elem(skel->maps.stats_map, &key, sizeof(key), &s, sizeof(s), 0) == 0) {
            if (s.count > 0) {
                printf("tcp_sendmsg: calls=%llu, avg=%llu ns, max=%llu ns, min=%llu ns\n",
                    s.count,
                    s.total_ns / s.count,
                    s.max_ns,
                    s.min_ns);
            }
        }
    }

cleanup:
    fentry_example_bpf__destroy(skel);
    return err < 0 ? -err : 0;
}
```

## Section 4: BPF_MODIFY_RETURN for Function Overriding

BPF_MODIFY_RETURN allows BPF programs to change the return value of a kernel function, effectively overriding its behavior without kernel patches.

```c
// modify_return_example.bpf.c
// Override security_file_open to deny access to specific files
#include "vmlinux.h"
#include <bpf/bpf_helpers.h>
#include <bpf/bpf_tracing.h>
#include <bpf/bpf_core_read.h>

// Map of denied file path prefixes
struct {
    __uint(type, BPF_MAP_TYPE_HASH);
    __uint(max_entries, 256);
    __type(key, char[64]);
    __type(value, __u8);
} denied_paths SEC(".maps");

// Event ring buffer for logging
struct {
    __uint(type, BPF_MAP_TYPE_RINGBUF);
    __uint(max_entries, 1 << 24);  // 16MB
} events SEC(".maps");

struct event {
    __u32 pid;
    __u32 uid;
    char comm[16];
    char filename[256];
    __u8 denied;
};

// BPF_MODIFY_RETURN: intercept security_file_open
// If we return non-zero, the original function is NOT called
SEC("fmod_ret/security_file_open")
int BPF_PROG(fmod_ret_security_file_open, struct file *file, int ret)
{
    // Only intercept if the original check passed
    if (ret != 0)
        return ret;

    struct dentry *dentry = BPF_CORE_READ(file, f_path.dentry);
    if (!dentry)
        return ret;

    // Read filename
    char filename[256] = {};
    struct qstr d_name = BPF_CORE_READ(dentry, d_name);
    bpf_probe_read_kernel_str(filename, sizeof(filename), d_name.name);

    // Check against denied paths
    char key[64] = {};
    __builtin_memcpy(key, filename, sizeof(key) - 1);

    __u8 *denied = bpf_map_lookup_elem(&denied_paths, &key);
    if (denied && *denied) {
        // Submit event to ring buffer
        struct event *e = bpf_ringbuf_reserve(&events, sizeof(*e), 0);
        if (e) {
            e->pid = bpf_get_current_pid_tgid() >> 32;
            e->uid = bpf_get_current_uid_gid() & 0xFFFFFFFF;
            bpf_get_current_comm(e->comm, sizeof(e->comm));
            __builtin_memcpy(e->filename, filename, sizeof(e->filename));
            e->denied = 1;
            bpf_ringbuf_submit(e, 0);
        }
        // Return -EPERM to deny access
        return -1; // -EPERM
    }

    return ret;  // Allow access
}
```

## Section 5: Performance Benchmarks: kprobe vs fentry

### Benchmark Methodology

```c
// benchmark.bpf.c - comparative overhead measurement
#include "vmlinux.h"
#include <bpf/bpf_helpers.h>
#include <bpf/bpf_tracing.h>

struct latency_key {
    __u32 probe_type;  // 0=kprobe, 1=fentry
    __u32 bucket;      // latency bucket in ns
};

struct {
    __uint(type, BPF_MAP_TYPE_HASH);
    __uint(max_entries, 1024);
    __type(key, struct latency_key);
    __type(value, __u64);
} latency_hist SEC(".maps");

struct {
    __uint(type, BPF_MAP_TYPE_PERCPU_ARRAY);
    __uint(max_entries, 2);
    __type(key, __u32);
    __type(value, __u64);
} timestamps SEC(".maps");

// kprobe version
SEC("kprobe/do_nanosleep")
int kprobe_do_nanosleep(struct pt_regs *ctx)
{
    __u32 key = 0;
    __u64 ts = bpf_ktime_get_ns();
    bpf_map_update_elem(&timestamps, &key, &ts, BPF_ANY);
    return 0;
}

SEC("kretprobe/do_nanosleep")
int kretprobe_do_nanosleep(struct pt_regs *ctx)
{
    __u32 key = 0;
    __u64 *start = bpf_map_lookup_elem(&timestamps, &key);
    if (!start) return 0;

    __u64 duration = bpf_ktime_get_ns() - *start;

    // Record in kprobe latency bucket
    struct latency_key lk = {.probe_type = 0, .bucket = duration / 1000};  // us buckets
    __u64 one = 1;
    __u64 *count = bpf_map_lookup_elem(&latency_hist, &lk);
    if (count) {
        __sync_fetch_and_add(count, 1);
    } else {
        bpf_map_update_elem(&latency_hist, &lk, &one, BPF_NOEXIST);
    }
    return 0;
}

// fentry version - identical logic, different attachment
SEC("fentry/do_nanosleep")
int BPF_PROG(fentry_do_nanosleep, const struct hrtimer_sleeper *t, enum hrtimer_mode mode)
{
    __u32 key = 1;
    __u64 ts = bpf_ktime_get_ns();
    bpf_map_update_elem(&timestamps, &key, &ts, BPF_ANY);
    return 0;
}

SEC("fexit/do_nanosleep")
int BPF_PROG(fexit_do_nanosleep, const struct hrtimer_sleeper *t, enum hrtimer_mode mode, int ret)
{
    __u32 key = 1;
    __u64 *start = bpf_map_lookup_elem(&timestamps, &key);
    if (!start) return 0;

    __u64 duration = bpf_ktime_get_ns() - *start;

    struct latency_key lk = {.probe_type = 1, .bucket = duration / 1000};
    __u64 one = 1;
    __u64 *count = bpf_map_lookup_elem(&latency_hist, &lk);
    if (count) {
        __sync_fetch_and_add(count, 1);
    } else {
        bpf_map_update_elem(&latency_hist, &lk, &one, BPF_NOEXIST);
    }
    return 0;
}
```

## Section 6: Go Function Tracing with uprobe and Trampoline

### Tracing Go Runtime Functions

Go's runtime functions are written in assembly and use a non-standard calling convention that complicates uprobe-based tracing. The BPF trampoline approach handles this more gracefully.

```c
// go_trace.bpf.c - Trace Go HTTP server function calls
#include "vmlinux.h"
#include <bpf/bpf_helpers.h>
#include <bpf/bpf_tracing.h>
#include <bpf/bpf_core_read.h>

// Map to store goroutine start times keyed by goroutine ID
struct {
    __uint(type, BPF_MAP_TYPE_HASH);
    __uint(max_entries, 65536);
    __type(key, __u64);   // goroutine ID
    __type(value, __u64); // start time ns
} goroutine_starts SEC(".maps");

// Ring buffer for HTTP request events
struct {
    __uint(type, BPF_MAP_TYPE_RINGBUF);
    __uint(max_entries, 1 << 26);  // 64MB
} http_events SEC(".maps");

struct http_event {
    __u64 timestamp_ns;
    __u64 duration_ns;
    __u32 pid;
    __u32 goroutine_id;
    char  method[8];
    char  path[256];
    __u32 status_code;
};

// Attach to net/http.(*ServeMux).ServeHTTP entry
// Use uprobe with the binary offset
SEC("uprobe/ServeHTTP")
int probe_serve_http(struct pt_regs *ctx)
{
    // In Go 1.17+ with register-based calling convention:
    // Arguments are in registers AX, BX, CX, DI, SI, R8, R9, R10, R11
    // For methods: receiver in AX, first arg in BX, etc.

    __u64 goid = 0;  // Extract goroutine ID from g struct
    // g pointer is in the FS register on Linux/amd64
    // g.goid is at offset 152 in Go 1.21
    __u64 g_ptr = 0;
    // bpf_probe_read_user reads from user space
    // In practice, use BCC or bpftrace for complex Go introspection

    __u64 pid_tgid = bpf_get_current_pid_tgid();
    __u64 ts = bpf_ktime_get_ns();

    bpf_map_update_elem(&goroutine_starts, &pid_tgid, &ts, BPF_ANY);
    return 0;
}

SEC("uretprobe/ServeHTTP")
int probe_serve_http_ret(struct pt_regs *ctx)
{
    __u64 pid_tgid = bpf_get_current_pid_tgid();
    __u64 *start = bpf_map_lookup_elem(&goroutine_starts, &pid_tgid);
    if (!start) return 0;

    __u64 duration = bpf_ktime_get_ns() - *start;
    bpf_map_delete_elem(&goroutine_starts, &pid_tgid);

    struct http_event *e = bpf_ringbuf_reserve(&http_events, sizeof(*e), 0);
    if (!e) return 0;

    e->timestamp_ns = bpf_ktime_get_ns();
    e->duration_ns = duration;
    e->pid = pid_tgid >> 32;
    bpf_ringbuf_submit(e, 0);

    return 0;
}
```

### Go Tool for Automated Symbol Resolution

```go
// gotracer/main.go - Automated Go function tracer using BPF trampolines
package main

import (
    "debug/elf"
    "debug/gosym"
    "fmt"
    "os"
    "strings"
)

// GoSymbolResolver resolves Go function names to binary offsets.
type GoSymbolResolver struct {
    symTable *gosym.Table
    elfFile  *elf.File
}

// NewGoSymbolResolver creates a resolver for the given binary.
func NewGoSymbolResolver(binaryPath string) (*GoSymbolResolver, error) {
    f, err := elf.Open(binaryPath)
    if err != nil {
        return nil, fmt.Errorf("failed to open ELF file: %w", err)
    }

    // Find Go symbol table sections
    pclnSection := f.Section(".gopclntab")
    if pclnSection == nil {
        return nil, fmt.Errorf("no .gopclntab section found (not a Go binary?)")
    }

    pclnData, err := pclnSection.Data()
    if err != nil {
        return nil, fmt.Errorf("failed to read .gopclntab: %w", err)
    }

    symSection := f.Section(".gosymtab")
    var symData []byte
    if symSection != nil {
        symData, err = symSection.Data()
        if err != nil {
            return nil, fmt.Errorf("failed to read .gosymtab: %w", err)
        }
    }

    lineTable := gosym.NewLineTable(pclnData, f.Section(".text").Addr)
    table, err := gosym.NewTable(symData, lineTable)
    if err != nil {
        return nil, fmt.Errorf("failed to parse Go symbol table: %w", err)
    }

    return &GoSymbolResolver{symTable: table, elfFile: f}, nil
}

// FindFunction returns the offset for a Go function name.
func (r *GoSymbolResolver) FindFunction(name string) (uint64, error) {
    for _, fn := range r.symTable.Funcs {
        if fn.Name == name || strings.HasSuffix(fn.Name, "."+name) {
            return fn.Entry, nil
        }
    }
    return 0, fmt.Errorf("function %q not found", name)
}

// ListHTTPHandlers lists all HTTP handler functions in the binary.
func (r *GoSymbolResolver) ListHTTPHandlers() []string {
    var handlers []string
    for _, fn := range r.symTable.Funcs {
        if strings.Contains(fn.Name, "http") &&
            (strings.Contains(fn.Name, "Handler") ||
             strings.Contains(fn.Name, "ServeHTTP")) {
            handlers = append(handlers, fn.Name)
        }
    }
    return handlers
}

func main() {
    if len(os.Args) < 2 {
        fmt.Fprintf(os.Stderr, "Usage: %s <binary>\n", os.Args[0])
        os.Exit(1)
    }

    resolver, err := NewGoSymbolResolver(os.Args[1])
    if err != nil {
        fmt.Fprintf(os.Stderr, "Error: %v\n", err)
        os.Exit(1)
    }

    fmt.Println("HTTP handlers found:")
    for _, h := range resolver.ListHTTPHandlers() {
        offset, _ := resolver.FindFunction(h)
        fmt.Printf("  %s @ 0x%x\n", h, offset)
    }
}
```

## Section 7: Production Use Cases

### Case Study 1: Database Query Latency Tracking

```c
// mysql_latency.bpf.c - Track MySQL query execution time
#include "vmlinux.h"
#include <bpf/bpf_helpers.h>
#include <bpf/bpf_tracing.h>

struct {
    __uint(type, BPF_MAP_TYPE_PERCPU_HASH);
    __uint(max_entries, 4096);
    __type(key, __u32);   // TID
    __type(value, __u64); // start time
} query_start SEC(".maps");

struct query_event {
    __u64 duration_ns;
    __u32 pid;
    __u32 tid;
    char  query[512];
    int   affected_rows;
};

struct {
    __uint(type, BPF_MAP_TYPE_RINGBUF);
    __uint(max_entries, 1 << 26);
} query_events SEC(".maps");

// Attach to mysqld dispatch_command function
// Use: uprobe:/usr/sbin/mysqld:*dispatch_command*
SEC("uprobe/dispatch_command")
int probe_dispatch_command(struct pt_regs *ctx)
{
    __u32 tid = bpf_get_current_pid_tgid() & 0xFFFFFFFF;
    __u64 ts = bpf_ktime_get_ns();
    bpf_map_update_elem(&query_start, &tid, &ts, BPF_ANY);
    return 0;
}

SEC("uretprobe/dispatch_command")
int probe_dispatch_command_ret(struct pt_regs *ctx)
{
    __u32 tid = bpf_get_current_pid_tgid() & 0xFFFFFFFF;
    __u64 *start = bpf_map_lookup_elem(&query_start, &tid);
    if (!start) return 0;

    __u64 duration = bpf_ktime_get_ns() - *start;
    bpf_map_delete_elem(&query_start, &tid);

    // Only record queries over 10ms
    if (duration < 10000000) return 0;

    struct query_event *e = bpf_ringbuf_reserve(&query_events, sizeof(*e), 0);
    if (!e) return 0;

    e->duration_ns = duration;
    e->pid = bpf_get_current_pid_tgid() >> 32;
    e->tid = tid;
    bpf_ringbuf_submit(e, 0);

    return 0;
}
```

```python
sched_latency.py - Measure scheduler run queue latency using fentry.

import ctypes

#include <uapi/linux/ptrace.h>
#include <linux/sched.h>

// fentry equivalent in BCC style
TRACEPOINT_PROBE(sched, sched_wakeup)

    u32 pid = args->pid;
    u64 ts = bpf_ktime_get_ns();

TRACEPOINT_PROBE(sched, sched_switch)
{

        u64 *ts = enqueue_time.lookup(&prev_pid);
        if (ts) {
            u64 latency_us = (bpf_ktime_get_ns() - *ts) / 1000;
            run_latency_us.increment(bpf_log2l(latency_us));
            enqueue_time.delete(&prev_pid);
        }
    }

    u32 next_pid = args->next_pid;
    u64 *ts = enqueue_time.lookup(&next_pid);
    if (ts) {
        u64 latency_us = (bpf_ktime_get_ns() - *ts) / 1000;
        run_latency_us.increment(bpf_log2l(latency_us));
        enqueue_time.delete(&next_pid);
    }
    return 0;
}
"""

b = BPF(text=bpf_text)
```

### Integrating with Prometheus

```go
// bpfmetrics/collector.go - Export BPF trampoline metrics to Prometheus
package bpfmetrics

import (
    "encoding/binary"
    "fmt"
    "unsafe"

    "github.com/cilium/ebpf"
    "github.com/prometheus/client_golang/prometheus"
)

// BPFMetricsCollector collects metrics from BPF maps.
type BPFMetricsCollector struct {
    statsMap *ebpf.Map

    callsTotal    *prometheus.Desc
    latencyAvgNs  *prometheus.Desc
    latencyMaxNs  *prometheus.Desc
}

// NewBPFMetricsCollector creates a new collector for the given BPF map.
func NewBPFMetricsCollector(statsMap *ebpf.Map, functionName string) *BPFMetricsCollector {
    labels := []string{"function"}
    return &BPFMetricsCollector{
        statsMap: statsMap,
        callsTotal: prometheus.NewDesc(
            "bpf_traced_calls_total",
            "Total number of traced function calls.",
            labels, nil,
        ),
        latencyAvgNs: prometheus.NewDesc(
            "bpf_traced_latency_avg_nanoseconds",
            "Average latency of traced function calls in nanoseconds.",
            labels, nil,
        ),
        latencyMaxNs: prometheus.NewDesc(
            "bpf_traced_latency_max_nanoseconds",
            "Maximum latency of traced function calls in nanoseconds.",
            labels, nil,
        ),
    }
}

func (c *BPFMetricsCollector) Describe(ch chan<- *prometheus.Desc) {
    ch <- c.callsTotal
    ch <- c.latencyAvgNs
    ch <- c.latencyMaxNs
}

type bpfStats struct {
    Count   uint64
    TotalNs uint64
    MaxNs   uint64
    MinNs   uint64
}

func (c *BPFMetricsCollector) Collect(ch chan<- prometheus.Metric) {
    key := uint32(0)
    var stats bpfStats

    if err := c.statsMap.Lookup(unsafe.Pointer(&key), unsafe.Pointer(&stats)); err != nil {
        return
    }

    if stats.Count == 0 {
        return
    }

    avgNs := stats.TotalNs / stats.Count
    _ = binary.LittleEndian // suppress import

    ch <- prometheus.MustNewConstMetric(
        c.callsTotal, prometheus.CounterValue,
        float64(stats.Count), "tcp_sendmsg",
    )
    ch <- prometheus.MustNewConstMetric(
        c.latencyAvgNs, prometheus.GaugeValue,
        float64(avgNs), "tcp_sendmsg",
    )
    ch <- prometheus.MustNewConstMetric(
        c.latencyMaxNs, prometheus.GaugeValue,
        float64(stats.MaxNs), "tcp_sendmsg",
    )
}

// Register registers the collector with the default Prometheus registry.
func (c *BPFMetricsCollector) Register() error {
    return prometheus.Register(c)
}
```
