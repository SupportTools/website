eBPF programs that read kernel data structures directly have historically been tied to the exact kernel version they were compiled against. BTF (BPF Type Format) and CO-RE (Compile Once — Run Everywhere) solve this by embedding type information into the kernel and using it to relocate field offsets at load time. This guide covers the complete BTF/CO-RE development workflow.

<!--more-->

## Section 1: What Is BTF and Why It Matters

BTF is a compact binary format that describes C types — structs, unions, enums, typedefs, and function signatures — embedded into ELF objects, kernel images, and BPF programs. It was originally designed to provide rich type information to the BPF verifier, but it became the foundation for CO-RE.

### The Problem BTF Solves

Before BTF, an eBPF program accessing `task_struct->pid` compiled with:

```c
// Hard-coded offset — breaks across kernel versions.
__u32 pid = *((__u32 *)(task + offsetof(struct task_struct, pid)));
```

This offset changes between kernel versions (and even between kernel configs). Programs compiled against Linux 5.15 break on Linux 6.1.

BTF + CO-RE changes this to:

```c
// CO-RE relocation — libbpf patches this offset at load time.
__u32 pid = BPF_CORE_READ(task, pid);
```

libbpf reads BTF from both the compiled program and the running kernel, computes the correct offset for the current kernel, and patches the instruction before loading.

### Checking Kernel BTF Support

```bash
# BTF is available if CONFIG_DEBUG_INFO_BTF=y was set at compile time.
ls /sys/kernel/btf/vmlinux

# Kernel must be >= 5.8 for full CO-RE support.
uname -r

# Check BTF features in bpftool.
bpftool feature probe | grep -i btf
```

## Section 2: Generating vmlinux.h

`vmlinux.h` is a single header file containing all BTF type definitions for the running kernel. It replaces dozens of kernel headers and ensures types match the currently running kernel.

```bash
# Generate vmlinux.h from the running kernel's BTF.
bpftool btf dump file /sys/kernel/btf/vmlinux format c > vmlinux.h

# Verify the output — it should be hundreds of thousands of lines.
wc -l vmlinux.h

# For cross-compilation, use BTF from a specific kernel image.
bpftool btf dump file /boot/vmlinuz-$(uname -r) format c > vmlinux.h
```

The generated header begins like:

```c
/* SPDX-License-Identifier: (LGPL-2.1 OR BSD-2-Clause) */
/* THIS FILE IS AUTOGENERATED! */
#ifndef __VMLINUX_H__
#define __VMLINUX_H__

#ifndef BPF_NO_PRESERVE_ACCESS_INDEX
#pragma clang attribute push (__attribute__((preserve_access_index)), \
    apply_to = record)
#endif
// ... thousands of struct definitions follow
```

The `preserve_access_index` attribute tells Clang to emit CO-RE relocation records for every field access.

## Section 3: Writing CO-RE eBPF Programs

### Project Structure

```
myebpf/
├── myebpf.bpf.c      # eBPF kernel-space program
├── myebpf.c          # User-space loader
├── myebpf.h          # Shared types between kernel and user space
├── vmlinux.h         # Generated from running kernel
└── Makefile
```

### Kernel-Space Program with CO-RE

```c
// myebpf.bpf.c
#include "vmlinux.h"
#include <bpf/bpf_helpers.h>
#include <bpf/bpf_core_read.h>
#include <bpf/bpf_tracing.h>

#include "myebpf.h"

// Map to send events to user space via ring buffer.
struct {
    __uint(type, BPF_MAP_TYPE_RINGBUF);
    __uint(max_entries, 1 << 24); // 16 MB
} events SEC(".maps");

// Map for per-CPU statistics.
struct {
    __uint(type, BPF_MAP_TYPE_PERCPU_ARRAY);
    __uint(max_entries, 1);
    __type(key, __u32);
    __type(value, __u64);
} syscall_count SEC(".maps");

// Tracepoint on sys_enter_execve.
SEC("tracepoint/syscalls/sys_enter_execve")
int trace_execve(struct trace_event_raw_sys_enter *ctx)
{
    struct event *e;
    struct task_struct *task;
    __u32 key = 0;
    __u64 *count;

    // Increment per-CPU counter.
    count = bpf_map_lookup_elem(&syscall_count, &key);
    if (count)
        __sync_fetch_and_add(count, 1);

    // Reserve space in the ring buffer.
    e = bpf_ringbuf_reserve(&events, sizeof(*e), 0);
    if (!e)
        return 0;

    // CO-RE: read task_struct fields — works across kernel versions.
    task = (struct task_struct *)bpf_get_current_task();

    e->pid  = BPF_CORE_READ(task, pid);
    e->tgid = BPF_CORE_READ(task, tgid);
    e->uid  = BPF_CORE_READ(task, cred, uid.val);

    // Read the parent PID via nested CO-RE access.
    e->ppid = BPF_CORE_READ(task, real_parent, pid);

    bpf_get_current_comm(e->comm, sizeof(e->comm));

    // Read the filename argument from the tracepoint context.
    bpf_probe_read_user_str(e->filename, sizeof(e->filename),
        (void *)ctx->args[0]);

    e->ts_ns = bpf_ktime_get_ns();

    bpf_ringbuf_submit(e, 0);
    return 0;
}

char LICENSE[] SEC("license") = "GPL";
```

### Shared Header

```c
// myebpf.h — types shared between kernel and user space.
#pragma once

#define TASK_COMM_LEN 16
#define MAX_FILENAME_LEN 127

struct event {
    __u32 pid;
    __u32 tgid;
    __u32 uid;
    __u32 ppid;
    char  comm[TASK_COMM_LEN];
    char  filename[MAX_FILENAME_LEN + 1];
    __u64 ts_ns;
};
```

## Section 4: CO-RE Read Macros

libbpf provides several macros for safe kernel memory access with CO-RE relocations:

```c
// Basic field read with CO-RE relocation.
u32 val = BPF_CORE_READ(task, field);

// Nested field read.
u32 uid = BPF_CORE_READ(task, cred, uid.val);

// Read into a local variable.
u64 start_time;
BPF_CORE_READ_INTO(&start_time, task, start_time);

// Read a bitfield — CO-RE handles bitfield offsets and shifts.
u8 in_execve = BPF_CORE_READ_BITFIELD(task, in_execve);

// Conditional existence check — returns true if the field exists on current kernel.
if (bpf_core_field_exists(task->io_uring))
    io_uring_ptr = BPF_CORE_READ(task, io_uring);

// Type existence check — program handles kernels without the type.
if (bpf_core_type_exists(struct mmap_lock))
    BPF_CORE_READ(mm, mmap_lock);
```

## Section 5: Compiling eBPF Programs

```makefile
# Makefile for CO-RE eBPF program

CLANG    ?= clang
BPFTOOL  ?= bpftool
CC       ?= gcc
ARCH     := $(shell uname -m | sed 's/x86_64/x86/' | sed 's/aarch64/arm64/')

CFLAGS   := -g -O2 -Wall -target bpf -D__TARGET_ARCH_$(ARCH)
INCLUDES := -I. -I/usr/include/bpf

.PHONY: all clean vmlinux

all: myebpf

# Step 1: Generate vmlinux.h from the running kernel.
vmlinux.h:
	$(BPFTOOL) btf dump file /sys/kernel/btf/vmlinux format c > $@

# Step 2: Compile the eBPF kernel program to an object file.
myebpf.bpf.o: myebpf.bpf.c vmlinux.h
	$(CLANG) $(CFLAGS) $(INCLUDES) -c $< -o $@

# Step 3: Generate the skeleton header for the user-space loader.
myebpf.skel.h: myebpf.bpf.o
	$(BPFTOOL) gen skeleton $< > $@

# Step 4: Compile the user-space loader.
myebpf: myebpf.c myebpf.skel.h
	$(CC) -g -O2 -Wall $(INCLUDES) $< -lbpf -lelf -lz -o $@

clean:
	rm -f myebpf myebpf.bpf.o myebpf.skel.h
```

```bash
# Install dependencies.
apt-get install -y clang llvm libelf-dev libbpf-dev linux-tools-$(uname -r)

# Build.
make

# Run (requires root or CAP_BPF).
sudo ./myebpf
```

## Section 6: User-Space Loader with Skeleton API

libbpf's skeleton API eliminates boilerplate for loading and attaching programs:

```c
// myebpf.c — user-space loader using the generated skeleton.
#include <stdio.h>
#include <stdlib.h>
#include <signal.h>
#include <unistd.h>
#include <bpf/libbpf.h>
#include "myebpf.skel.h"
#include "myebpf.h"

static volatile bool running = true;

static void sig_handler(int sig) { running = false; }

static int handle_event(void *ctx, void *data, size_t sz) {
    const struct event *e = data;
    printf("%-6u %-6u %-6u %-16s %s\n",
        e->pid, e->tgid, e->ppid, e->comm, e->filename);
    return 0;
}

int main(void) {
    struct myebpf_bpf *skel = NULL;
    struct ring_buffer *rb = NULL;
    int err;

    signal(SIGINT, sig_handler);
    signal(SIGTERM, sig_handler);

    // Set libbpf logging to stderr.
    libbpf_set_print(NULL);

    // Open the BPF object — loads BTF and prepares CO-RE relocations.
    skel = myebpf_bpf__open();
    if (!skel) {
        fprintf(stderr, "Failed to open BPF skeleton\n");
        return 1;
    }

    // Load and verify — applies CO-RE relocations and loads maps.
    err = myebpf_bpf__load(skel);
    if (err) {
        fprintf(stderr, "Failed to load BPF skeleton: %d\n", err);
        goto cleanup;
    }

    // Attach all programs defined with SEC() attributes.
    err = myebpf_bpf__attach(skel);
    if (err) {
        fprintf(stderr, "Failed to attach BPF programs: %d\n", err);
        goto cleanup;
    }

    // Create a ring buffer consumer.
    rb = ring_buffer__new(bpf_map__fd(skel->maps.events),
        handle_event, NULL, NULL);
    if (!rb) {
        fprintf(stderr, "Failed to create ring buffer\n");
        err = -1;
        goto cleanup;
    }

    printf("%-6s %-6s %-6s %-16s %s\n", "PID", "TGID", "PPID", "COMM", "FILENAME");

    while (running) {
        err = ring_buffer__poll(rb, 100 /* timeout ms */);
        if (err < 0 && err != -EINTR) {
            fprintf(stderr, "ring buffer poll error: %d\n", err);
            break;
        }
    }

cleanup:
    ring_buffer__free(rb);
    myebpf_bpf__destroy(skel);
    return err < 0 ? -err : 0;
}
```

## Section 7: BTF Inspection with bpftool

bpftool provides comprehensive BTF inspection capabilities:

```bash
# List all BTF objects loaded in the kernel.
bpftool btf list

# Show the BTF for a specific BPF program.
bpftool btf show id <btf-id>

# Dump full type information for a struct.
bpftool btf dump id <btf-id> format c | grep -A 30 "struct task_struct {"

# Inspect BTF from a compiled eBPF object.
bpftool btf dump file myebpf.bpf.o

# Show BPF maps with BTF type info.
bpftool map list

# Show loaded BPF programs with their BTF.
bpftool prog list

# Dump a program's instructions with BTF-enriched annotations.
bpftool prog dump xlated id <prog-id> linum

# Show CO-RE relocations in a compiled object.
llvm-objdump --section=.BTF.ext -d myebpf.bpf.o

# Verify BTF correctness.
bpftool btf dump file myebpf.bpf.o | head -50
```

## Section 8: Handling Kernel Version Differences with CO-RE

CO-RE provides several mechanisms for handling types and fields that differ or are missing across kernel versions.

### Conditional Field Access

```c
// Check if a field exists before accessing it.
// This compiles to a no-op on kernels where the field exists;
// it skips the access on kernels where it doesn't.
struct task_struct *task = (void *)bpf_get_current_task();

// __builtin_preserve_field_info is the low-level primitive behind CO-RE macros.
if (bpf_core_field_exists(task->in_execve)) {
    u8 in_execve = BPF_CORE_READ_BITFIELD(task, in_execve);
    // use in_execve
}
```

### Type Compatibility Checks

```c
// Handle struct layout changes between kernel versions.
// bpf_core_type_size returns 0 if type doesn't exist.
if (bpf_core_type_size(struct mmap_lock) > 0) {
    // Newer kernel with mmap_lock.
} else {
    // Older kernel with mmap_sem.
}
```

### BPF_CORE_READ_STR_INTO for Strings

```c
char filename[256];
// Safely read a kernel string with CO-RE.
bpf_core_read_str(filename, sizeof(filename),
    &BPF_CORE_READ(task, mm, exe_file, f_path.dentry, d_name.name));
```

## Section 9: Distributing Portable eBPF Programs

The CO-RE workflow allows shipping a single eBPF binary that runs on any Linux kernel from 5.8 onwards:

```bash
# Build a statically-linked user-space loader with embedded BTF.
# The eBPF object is compiled-in via the skeleton's xxd embedding.
gcc -g -O2 -static myebpf.c -lbpf -lelf -lz -o myebpf-static

# Verify no dynamic dependencies.
ldd myebpf-static

# The binary includes the .bpf.o with full BTF — confirm.
readelf -S myebpf-static | grep bpf

# Test on a different kernel version without recompiling.
# Copy myebpf-static to a system running Linux 5.15, 6.1, or 6.6.
scp myebpf-static user@other-kernel:/tmp/
ssh user@other-kernel "sudo /tmp/myebpf-static"
```

### Container Distribution

Package portable eBPF programs in containers for Kubernetes DaemonSet deployment:

```dockerfile
FROM ubuntu:24.04 AS builder
RUN apt-get update && apt-get install -y \
    clang llvm libelf-dev libbpf-dev \
    linux-tools-generic make
WORKDIR /build
COPY . .
RUN make

FROM gcr.io/distroless/cc-debian12
COPY --from=builder /build/myebpf /myebpf
ENTRYPOINT ["/myebpf"]
```

The DaemonSet needs `privileged: true` (or specific capabilities) and the `/sys/kernel/btf/vmlinux` volume to load BTF at runtime when the embedded BTF differs from the kernel:

```yaml
apiVersion: apps/v1
kind: DaemonSet
metadata:
  name: myebpf-tracer
  namespace: kube-system
spec:
  selector:
    matchLabels:
      app: myebpf-tracer
  template:
    metadata:
      labels:
        app: myebpf-tracer
    spec:
      hostPID: true
      hostNetwork: true
      containers:
        - name: tracer
          image: registry.example.com/myebpf:v1.0.0
          securityContext:
            privileged: true
          volumeMounts:
            - name: sys
              mountPath: /sys
              readOnly: true
            - name: debugfs
              mountPath: /sys/kernel/debug
      volumes:
        - name: sys
          hostPath:
            path: /sys
        - name: debugfs
          hostPath:
            path: /sys/kernel/debug
```

BTF and CO-RE transformed eBPF from a compile-per-kernel discipline into a genuine portable platform. The investment in learning the `BPF_CORE_READ` macros and the btf/skeleton workflow pays compound returns as your observability and security programs continue running through kernel upgrades without recompilation.
